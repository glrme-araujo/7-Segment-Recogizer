In[6]:
modelo = Sequential()                   # Criar modelo sequencial para arquitetura da rede

INPUT_SHAPE=(28, 28, 1)                 # Formato do dado de entrada


# Adicionar 1 Camada de convoluo de 32 
# 32 neurnios
# Kernel 3x3 
# Funo de Ativao = Relu
modelo.add(layers.Conv2D(32, (3, 3),
                         data_format = 'channels_last', 
                         activation='relu',
                         input_shape=INPUT_SHAPE))
# Pooling de 2x2
modelo.add(layers.MaxPooling2D((2, 2)))



# Adicionar 2 Camada de convoluo de 64 
# 32 neurnios
# Kernel 3x3 
# Funo de Ativao = Relu
modelo.add(layers.Conv2D(64, (3, 3),
                         data_format = 'channels_last', 
                         activation='relu',
                         input_shape=INPUT_SHAPE))

# Pooling de 2x2
modelo.add(layers.MaxPooling2D((2, 2)))


modelo.add(Flatten())          # Comprime a matriz 

modelo.add(layers.Dense(100))  # Diminuir a saida para 100 Neurnios


# Diminuir a saida para 10 Neurnios (10 Classes)
# Funa de Ativao Softmax 
modelo.add(layers.Dense(10, activation = 'softmax')) 

modelo.summary()
